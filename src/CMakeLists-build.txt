cmake_minimum_required(VERSION 3.16)

project(telecloud-multi VERSION 1.00 LANGUAGES CXX C)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

# Qt packages
find_package(Qt6 REQUIRED COMPONENTS 
    Core 
    Quick 
    Network 
    Concurrent
    QuickControls2
)

# FFmpeg paths (using downloaded prebuilt)
set(FFMPEG_ROOT "$ENV{HOME}/ffmpeg/ffmpeg-master-latest-linux64-gpl-shared")
set(FFMPEG_INCLUDE_DIR "${FFMPEG_ROOT}/include")
set(FFMPEG_LIB_DIR "${FFMPEG_ROOT}/lib")

# FFmpeg libraries
set(FFMPEG_LIBRARIES
    ${FFMPEG_LIB_DIR}/libavformat.so
    ${FFMPEG_LIB_DIR}/libavcodec.so
    ${FFMPEG_LIB_DIR}/libavutil.so
    ${FFMPEG_LIB_DIR}/libavdevice.so
    ${FFMPEG_LIB_DIR}/libavfilter.so
    ${FFMPEG_LIB_DIR}/libswscale.so
    ${FFMPEG_LIB_DIR}/libswresample.so
)

# Source files
set(SOURCES
    main.cpp
    core/errorlogger.h
    core/networkscanner.h
    core/networkscanner.cpp
    core/rtspprobe.h
    core/rtspprobe.cpp
    core/recorder.h
    core/recorder.cpp
    core/dvrmanager.h
    core/dvrmanager.cpp
)

# QML and resource files
set(QML_FILES
    ui/main.qml
    ui/styles/Theme.qml
    ui/pages/DashboardPage.qml
    ui/pages/ConfigurationPage.qml
    ui/components/Card.qml
)

# Resource file
qt_add_resources(QML_RESOURCES "resources.qrc"
    PREFIX "/"
    BASE "${CMAKE_SOURCE_DIR}"
    FILES
        ${QML_FILES}
        Icon.png
)

# Create executable
add_executable(telecloud-multi
    ${SOURCES}
    ${QML_RESOURCES}
)

# Include directories
target_include_directories(telecloud-multi PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}/core
    ${FFMPEG_INCLUDE_DIR}
)

# Link Qt libraries
target_link_libraries(telecloud-multi PRIVATE
    Qt6::Core
    Qt6::Quick
    Qt6::Network
    Qt6::Concurrent
    Qt6::QuickControls2
)

# Link FFmpeg libraries
target_link_libraries(telecloud-multi PRIVATE
    ${FFMPEG_LIBRARIES}
)

# Add FFmpeg library directories
target_link_directories(telecloud-multi PRIVATE
    ${FFMPEG_LIB_DIR}
)

# Compiler definitions
target_compile_definitions(telecloud-multi PRIVATE
    QT_MESSAGELOGCONTEXT
)

# Set RPATH for FFmpeg libraries
set_target_properties(telecloud-multi PROPERTIES
    BUILD_RPATH "${FFMPEG_LIB_DIR}"
    INSTALL_RPATH "${FFMPEG_LIB_DIR}"
)

# Print configuration summary
message(STATUS "=== TELECLOUD-MULTI Build Configuration ===")
message(STATUS "Version: ${PROJECT_VERSION}")
message(STATUS "Qt Version: ${Qt6_VERSION}")
message(STATUS "FFmpeg Include: ${FFMPEG_INCLUDE_DIR}")
message(STATUS "FFmpeg Lib: ${FFMPEG_LIB_DIR}")
message(STATUS "===========================================")
